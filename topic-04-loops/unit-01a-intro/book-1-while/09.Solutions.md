## Exercise 1 - Repeat a message

~~~python
name = input("Enter name: ")
times = int(input("How many times? "))

count = 1

while count <= times:
    print("Hello", name)
    count = count + 1
~~~

## Exercise 2 - Positive number validator

~~~python
def get_positive_number():
    number = int(input("Enter a positive number: "))

    while number <= 0:
        print("Invalid number")
        number = int(input("Enter a positive number: "))

    return number


value = get_positive_number()
print("You entered:", value)
~~~

## Exercise 4 - Count spaces

~~~python
sentence = input("Enter a sentence: ")

index = 0
space_count = 0

while index < len(sentence):
    if sentence[index] == " ":
        space_count = space_count + 1

    index = index + 1

print("Number of spaces:", space_count)
~~~

## Exercise 5 - Total until stop

~~~python
total = 0

number = int(input("Enter number, 0 to stop: "))

while number != 0:
    total = total + number
    number = int(input("Enter number, 0 to stop: "))

print("Total:", total)
~~~

## Exercise 6 - Menu program

~~~python
def print_welcome():
    print("Welcome to the program")


def double_number():
    number = int(input("Enter number: "))
    print("Doubled:", number * 2)


def check_even_odd():
    number = int(input("Enter number: "))

    if number % 2 == 0:
        print("Even")
    else:
        print("Odd")


choice = ""

while choice != "4":
    print("1. Print welcome message")
    print("2. Double a number")
    print("3. Check if a number is even or odd")
    print("4. Exit")

    choice = input("Enter choice: ")

    if choice == "1":
        print_welcome()
    elif choice == "2":
        double_number()
    elif choice == "3":
        check_even_odd()
    elif choice == "4":
        print("Goodbye")
    else:
        print("Invalid choice")
~~~

## Challenge 3 - Student mark system

~~~python
def get_valid_mark():
    mark = int(input("Enter mark: "))

    while mark < 0 or mark > 100:
        print("Invalid mark")
        mark = int(input("Enter mark: "))

    return mark


def get_grade(mark):
    if mark >= 70:
        return "Distinction"
    elif mark >= 60:
        return "Merit"
    elif mark >= 40:
        return "Pass"
    else:
        return "Fail"


mark = None
choice = ""

while choice != "3":
    print("1. Enter a valid mark")
    print("2. Show grade")
    print("3. Exit")

    choice = input("Enter choice: ")

    if choice == "1":
        mark = get_valid_mark()
    elif choice == "2":
        if mark is None:
            print("Please enter a mark first")
        else:
            print("Grade:", get_grade(mark))
    elif choice == "3":
        print("Goodbye")
    else:
        print("Invalid choice")
~~~

## Challenge 4 - Simple bank menu

~~~python
def get_positive_amount():
    amount = float(input("Enter amount: "))

    while amount <= 0:
        print("Amount must be positive")
        amount = float(input("Enter amount: "))

    return amount


balance = 0
choice = ""

while choice != "4":
    print("1. Deposit")
    print("2. Withdraw")
    print("3. View balance")
    print("4. Exit")

    choice = input("Enter choice: ")

    if choice == "1":
        amount = get_positive_amount()
        balance = balance + amount
        print("Deposit complete")
    elif choice == "2":
        amount = get_positive_amount()

        if amount <= balance:
            balance = balance - amount
            print("Withdrawal complete")
        else:
            print("Insufficient funds")
    elif choice == "3":
        print("Balance:", balance)
    elif choice == "4":
        print("Goodbye")
    else:
        print("Invalid choice")
~~~
