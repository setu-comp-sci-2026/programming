# Sample Solutions

These are sample solutions. Your own programs may look different and still be correct.

## Square

~~~python
import turtle

screen = turtle.Screen()
t = turtle.Turtle()

for side in range(4):
    t.forward(100)
    t.right(90)

turtle.done()
~~~

## Triangle

~~~python

~~~

## User Shape

~~~python
import turtle

sides = int(input("How many sides? "))
length = int(input("Side length? "))
colour = input("Colour? ")

screen = turtle.Screen()
t = turtle.Turtle()
t.color(colour)

angle = 360 / sides

for side in range(sides):
    t.forward(length)
    t.left(angle)

turtle.done()
~~~

## Flower

~~~python
import turtle

screen = turtle.Screen()
t = turtle.Turtle()
t.speed(0)
t.color("purple")

for petal in range(18):
    t.circle(80)
    t.left(20)

turtle.done()
~~~

## Spiral

~~~python
import turtle

screen = turtle.Screen()
t = turtle.Turtle()
t.speed(0)

length = 5

for line in range(60):
    t.forward(length)
    t.right(90)
    length += 5

turtle.done()
~~~
